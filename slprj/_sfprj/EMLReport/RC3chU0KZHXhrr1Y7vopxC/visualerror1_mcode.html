1function [A, error, d_h] = visualerror(p, ps)
 2%% get the visual error message
 3% cal s and s*
 4[m,n] = size(p);
 5s = (1/500)*(p - repmat([640;400],[1 n]));
 6s = [s; ones(1,n)];
 7for i=1:n
 8    s(:,i) = s(:,i)./norm(s(:,i));
 9end
10st = (1/500)*(ps - repmat([640;400],[1 n]));
11st = [st; ones(1,n)];
12for i=1:n
13    st(:,i) = st(:,i)./norm(st(:,i));
14end
15% cal sum of s
16h = sum(s,2);
17ht = sum(st,2);
18% delta h 
19delta_h = h - ht;
20d_h = delta_h;
21%% modefied delta h
22namda = 2;
23ht_stat = ht./norm(ht);
24% error = h - ht;
25error = cross(ht_stat, h) + namda*ht_stat*(ht_stat'*delta_h);
26%% A for use
27sk_ht = [0 -ht_stat(3) ht_stat(2);...
28    ht_stat(3) 0 -ht_stat(1);...
29    -ht_stat(2) ht_stat(1) 0];
30A = -sk_ht + namda*(ht_stat*ht_stat');
