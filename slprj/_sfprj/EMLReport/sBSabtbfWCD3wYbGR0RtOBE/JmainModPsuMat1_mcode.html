1function [Jpseudo, Nm] = JmainModPsuMat(Jmain, d_h)
 2% calculate the modified pseudo-inverse of main task jacobian matrix
 3% zhonghang
 4% 2016-12-23
 5%% define parametes
 6k = 0.1;
 7W1 = [eye(3) zeros([3 7]); zeros([7 3]) zeros([7 7])];
 8W2 = [zeros([6 6]) zeros([6 4]); zeros([4 6]) diag([1 1 1 1])];
 9D_H = 0.2;
10D_L = 0.1;
11%% calculate the main psudo inverse jacobian matrix
12% calculate the dynamic penalty parameter namda
13norm_dh = norm(d_h);
14if norm_dh <= D_L
15    namda = 1;
16elseif norm_dh >= D_H
17    namda = 0;
18else
19    namda = (D_H-norm_dh)/(D_H-D_L);
20end
21% calculate the modified pseduo-inverse of matrix
22Jpseudo = pinv(Jmain'*Jmain + k*namda*W1.^2 + k*(1-namda)*W2.^2) * Jmain';
23% calculate the null space of the jacobian matrix
24Nm = eye(10) - Jpseudo*Jmain;
